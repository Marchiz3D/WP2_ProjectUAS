try {
    const data = { email, password }
    const response = await axios.post(
      "http://localhost:5000/api/customers/login",
      data
    )
    if (response) {
      setIsLogin(true)
      setTimeout(() => {
        navigate("/kamar")
      }, 1000)
    } else {
      setIsLogin(false)
      setErrorMessage(response.data.message)
      console.log(response)
    }
  } catch (error) {
    setErrorMessage(error.response.data.message)
    console.log(error)
  }
